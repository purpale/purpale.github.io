import java.util.Scanner;

public class ZHtest1_jjj {
	public static void main(String[] args) {
		Scanner reader = new Scanner(System.in);
		System.out.print("体重（单位：千克）：");
		double ti3zhong4 = reader.nextDouble();
		System.out.print("身高（单位：米）：");
		double shen1gao1 = reader.nextDouble();
		
		double bmi = ti3zhong4 / (shen1gao1 * shen1gao1);
		System.out.println(bmi);
		
		if(bmi < 18.5) {
			System.out.println("消瘦");
		}
		else if(bmi < 24) {
			System.out.println("正常");
		}
		else if(bmi < 28) {
			System.out.println("偏胖");
		}
		else {
			System.out.println("肥胖");
		}
	}
}
